-- | `run` wraps some exprs with lambda redundant things from AST.
-- Specifically, this converts @if p then a else b@ to something like @(if p then (lambda. a) else (lambda. b))()@.
--
-- For example, this converts:
--
-- > let rec fact n =
-- >     if n == 0 then
-- >         1
-- >     else
-- >         n * fact (n - 1)
-- > in fact 10
--
-- to:
--
-- > let rec fact n =
-- >     (if n == 0 then
-- >         fun -> 1
-- >     else
-- >         fun -> n * fact (n - 1)
-- >     )()
-- > in fact 10
